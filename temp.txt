try {
    def subResults = prev.getSubResults()
    log.info("Sub-results class: " + subResults?.getClass()?.getName())
    
    // ConcurrentLinkedQueue - use iterator
    def iterator = subResults?.iterator()
    while (iterator != null && iterator.hasNext()) {
        def sub = iterator.next()
        def url = sub.getUrlAsString()
        log.info("Sub URL: " + url)
        if (url != null && url.contains("code=")) {
            def code = url.replaceAll(".*[?&]code=([^&]+).*", "\$1")
            vars.put("C_ACode", code)
            log.info("SUCCESS: " + code)
        }
    }
    
    // Also log direct methods
    log.info("Redirect: " + prev.getRedirectLocation())
    log.info("Main URL: " + prev.getUrlAsString())
    
} catch (Exception e) {
    log.error("Script error: " + e.getMessage())
}
